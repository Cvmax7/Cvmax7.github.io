#define _CRT_SECURE_NO_WARNINGS 1
#include <stdio.h>

int gcd(int a, int b);

int main()
{
    int a, b;
    printf("请输入a,b\n");
    scanf("%d,%d",&a,&b);
    printf("a和b最大公因数为：%d", gcd(a, b));
    return 0;
}

int gcd(int a, int b)
{
    int temp;
    if (a < b)
    {
        temp = b;
        b = a;
        a = temp;
    }
    while (a % b)
    {
        temp = b;
        b = a % b;
        a = temp;
    }
    return b;
}

3.2 迭代版本的egcd

#define _CRT_SECURE_NO_WARNINGS 1
#include <stdio.h>

// ra+sb = gcd(a,b)
void egcd(int a, int b);

int main()
{
    int a, b;
    printf("请输入a,b\n");
    scanf("%d,%d", &a, &b);
    egcd(a, b);
    return 0;
}

void egcd(int a, int b)
{
    int s1 = 1;
    int r0 = 1;
    int s0 = 0;
    int r1 = 0;
    int q;
    int temp;
    int res_r;
    int res_s;
    int res_gcd;
    while (b)
    {
        q = a / b;
        temp = a;
        a = b;
        b = temp % b;
        temp = r0;
        r0 = r1;
        r1 = temp - q * r1;
        temp = s0;
        s0 = s1;
        s1 = temp - q * s1;
    }
    res_r = r0;
    res_s = s0;
    res_gcd = a;
    printf("r=%d,s=%d,gcd(a,b)=%d", res_r, res_s, res_gcd);
}
